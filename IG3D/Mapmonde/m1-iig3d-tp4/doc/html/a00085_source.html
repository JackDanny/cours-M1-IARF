61 inline glm::vec3 perpendicular (const glm::vec3 & v) {
62  static const float fSquareZero = 1e-06 * 1e-06;
63  glm::vec3 perp = glm::cross(v, glm::vec3(1,0,0));
64  if (sqlength(perp)< fSquareZero) {
65  /* This vector is the Y axis multiplied by a scalar, so we have
66  to use another axis.
67  */
68  perp = glm::cross(v, glm::vec3(0,1,0));
69  }
70  return perp;
71 }
