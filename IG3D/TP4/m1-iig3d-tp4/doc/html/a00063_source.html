106 void Mesh::computeNormals (void) {
107  for (TriangleIndexArray::iterator f_iter = triangles_.begin() ; f_iter != triangles_.end() ; ++f_iter) {
108  Vertex& v0 = vertices_[f_iter->indexes_[0]];
109  Vertex& v1 = vertices_[f_iter->indexes_[1]];
110  Vertex& v2 = vertices_[f_iter->indexes_[2]];
111  glm::vec3 p0, p1, p2;
112  p0 = v0.position_;
113  p1 = v1.position_;
114  p2 = v2.position_;
115 
116  // compute normals
117  glm::vec3 f_normal;
118  f_normal = glm::cross((p1 - p0), (p2 - p0));
119  glm::normalize(f_normal);
120  v0.normal_ += f_normal;
121  v1.normal_ += f_normal;
122  v2.normal_ += f_normal;
123  }
124  for (VertexArray::iterator v_iter = vertices_.begin() ; v_iter != vertices_.end() ; ++v_iter) {
125  glm::normalize(v_iter->normal_);
126  }
127 }
