404 int Renderer::initShader(std::string vsFileName, std::string fsFileName)
405 {
406 
407  // Les opérations devant être faites pour pouvoir utiliser des shaders OpenGL dans notre application sont les suivantes :
408  // 1 - Vertex shader :
409  // 1.1 - Charger le source depuis le fichier "../shaders/vertexdefault.glsl" en utilisant la fonction loaders::text::loadFile() de lecture d'un fichier texte
410  char* vertexShaderSource = loaders::text::loadFile(vsFileName.c_str());
411  // 1.2 - Créer un objet OpenGL VERTEX_SHADER et y associer le source
412  int vertexShaderId;
413  glAssert(vertexShaderId = glCreateShader(GL_VERTEX_SHADER));
414  glAssert(glShaderSource(vertexShaderId, 1, (const GLchar **)&vertexShaderSource, NULL));
415  // 1.3 - Compiler le shader
416  glAssert(glCompileShader(vertexShaderId));
417  // 1.4 - Vérifier les erreurs de compilation
418  GLint compiled;
419  glAssert(glGetShaderiv(vertexShaderId, GL_COMPILE_STATUS, &compiled));
420  if (!compiled) {
421  std::cerr << "[" << vsFileName << "] Vertex shader not compiled : " << std::endl;
422  printShaderInfoLog(vertexShaderId);
423  return 0;
424  }
425 
426  // 2 - Fragment shader :
427  // 2.1 - Charger le source depuis le fichier "../shaders/fragmentdefault.glsl" en utilisant la fonction loaders::text::loadFile() de lecture d'un fichier texte
428  char *fragmentShaderSource;
429  fragmentShaderSource = loaders::text::loadFile(fsFileName.c_str());
430  // 2.2 - Créer un objet OpenGL FRAGMENT_SHADER et y associer le source
431  int fragmentShaderId;
432  glAssert(fragmentShaderId = glCreateShader(GL_FRAGMENT_SHADER));
433  glAssert(glShaderSource(fragmentShaderId, 1, (const GLchar **)&fragmentShaderSource, NULL));
434  // 2.3 - Compiler le shader
435  glAssert(glCompileShader(fragmentShaderId));
436  // 2.4 - Vérifier les erreurs de compilation
437  glAssert(glGetShaderiv(fragmentShaderId, GL_COMPILE_STATUS, &compiled));
438  if (!compiled) {
439  std::cerr << "[" << fsFileName << "] fragment shader not compiled : " << std::endl;
440  printShaderInfoLog(fragmentShaderId);
441  return 0;
442  }
443 
444  // 3 - Programme :
445  // 3.1 - Créer un programme OpenGL, stocker son identifiant dans g_Program et y associer les shaders
446  int program;
447  glAssert(program = glCreateProgram());
448  glAssert(glAttachShader(program, vertexShaderId));
449  glAssert(glAttachShader(program, fragmentShaderId));
450  // 3.2 - Fixer les positions des attributs géométriques en accord avec la classe MyGLMesh et les conventions
451  // d'écriture du source GLSL inPosition --> index 0, inNormal --> index 1 et inTexCoord --> index 2.
452  glAssert(glBindAttribLocation(program, 0, "inPosition"));
453  glAssert(glBindAttribLocation(program, 1, "inNormal"));
454  glAssert(glBindAttribLocation(program, 2, "inTexCoord"));
455  // 3.3 - Lier le programme
456  glAssert(glLinkProgram(program));
457  // 3.4 - Vérifier les erreurs d'édition de lien
458  GLint linked;
459  glAssert(glGetProgramiv(program, GL_LINK_STATUS, &linked));
460  if (!linked) {
461  std::cerr << "Program not linked" << std::endl;
462  printProgramInfoLog(program);
463  }
464  return program;
465

484  // 1 - Préparer de l'image pour le rendu :
485  // 1.1 - Effacer les buffers de destination
486  glAssert(glClearColor(0.f, 0.3f, 0.3f, 0.f));
487  glAssert(glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT));
488 
489  // 2 - Construire les matrices de vue et de projection :
490  // 2.1 - Définir une matrice de projection perspective

493  glm::mat4x4 localmodelViewMatrix = viewMatrix_;
494

496  // 3.1 - Activer le programme par défaut
497  glAssert(glUseProgram(program_));
498 
499  // 3.2 - Récupérer les adresses des paramètres
500  GLint transformationLoc[4];
501  glAssert(transformationLoc[0] = glGetUniformLocation(program_, "modelViewMatrix"));
502  glAssert(transformationLoc[1] = glGetUniformLocation(program_, "projectionMatrix"));
503  glAssert(transformationLoc[2] = glGetUniformLocation(program_, "normalMatrix"));
504  glAssert(transformationLoc[3] = glGetUniformLocation(program_, "MVP"));
505 
506  GLint materialLoc[4];

509  glAssert(materialLoc[2] = glGetUniformLocation(program_, "materialNs"));
510  glAssert(materialLoc[3] = glGetUniformLocation(program_, "baseTexture"));
511 
512  // 3.3 - positionnez les paramètres indépendants des objets (source de lumière, ...)
513 
514  // lights Binding
515  glAssert(glUniform3fv(glGetUniformLocation(program_, "keyLightPosition"), 1, glm::value_ptr(lights_[0]->getPosition())));
516  glAssert(glUniform3fv(glGetUniformLocation(program_, "keyLightColor"), 1, glm::value_ptr(lights_[0]->getColor())));

577 int Renderer::handleMouseEvent(const MouseEvent & event)
578 {
579  static int modifiers = 0;
580  static int x = 0;
581  static int y = 0;
582  static int button = 0;
583  static bool moved = false;
584 
585  if (event.click) {
586  x = event.x;
587  y = event.y;
588  button = event.button;
589  modifiers = event.modifiers;
590  }
591 
592  if (event.button == MouseEvent::MOVE) {
593  float dx = (float)(event.x - x) / (float) width_;
594  float dy = (float)(event.y - y) / (float) height_;
595  switch (button) {
596  case MouseEvent::LEFT : {
597  // Début du code à écrire
598  // 1 - Récupérer les axes xvec et yvec de la caméra
599  // 2 - Modifier g_viewMatrix pour y ajouter une rotation de dy*360.f degrés autour de xvec
600  // 3 - Modifier viewMatrix_ pour y ajouter une rotation de dx*360.f degrés autour de yvec
601  // Fin du code à écrire
602  glm::vec3 xvec = glm::vec3(glm::row(viewMatrix_, 0));
603  glm::vec3 yvec = glm::vec3(glm::row(viewMatrix_, 1));
604  viewMatrix_ = glm::rotate(viewMatrix_, dy * 360.f, xvec);
605  viewMatrix_ = glm::rotate(viewMatrix_, dx * 360.f, yvec);
606 
607  }
608  break;
609  case MouseEvent::RIGHT : {
610  // Début du code à écrire
611  // 1 - Calculer les vecteurs déplacement xvec et yvec selon les axes X et Y de la caméra
612  // 2 - Modifier viewMatrix_ pour y ajouter une translation de xvec+yvec
613  // Fin du code à écrire
614  glm::vec3 xvec = glm::vec3(glm::row(viewMatrix_, 0)) * dx * 10.f;
615  glm::vec3 yvec = glm::vec3(glm::row(viewMatrix_, 1)) * dy * 10.f;
616  glm::vec3 tvec = xvec + yvec;
617  viewMatrix_ = glm::translate(viewMatrix_, tvec);
618 
619  }
620  break;
621  case MouseEvent::MIDDLE : {
622  // Début du code à écrire
623  // 1 - Calculer le vecteur déplacement zvec selon l'axe Z de la caméra
624  // 2 - Modifier viewMatrix_ pour y ajouter une translation de zvec
625  // Fin du code à écrire
626  glm::vec3 zvec = glm::vec3(glm::row(viewMatrix_, 2)) * dy * 10.f;
627  viewMatrix_ = glm::translate(viewMatrix_, zvec);
628 
629  }
630  break;
631  }
632  x = event.x;
633  y = event.y;
634  moved = true;
635  return 1;
636  } else {
637  moved = false;
638  }
639 
640  if (event.button == MouseEvent::WHEEL) {
641  return 1;
642  }
643 
644  if (!event.click) {
645  button = 0;
646  modifiers = 0;
647  }
648 
649  return 1;
650 }
