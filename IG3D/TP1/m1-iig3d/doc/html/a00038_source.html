37 char* loadFile(const char *fileName)
38 {
39  std::ifstream file;
40  file.open(fileName, std::ios::in);
41  if (!file) {
42  std::cerr << "could not open file " << fileName << std::endl;
43  char *shaderSource = new char[1];
44  shaderSource[0]=0;
45  return shaderSource;
46  }
47 
48  unsigned long len = getFileLength(file);
49  if (len == 0) {
50  char *shaderSource = new char[1];
51  shaderSource[0]=0;
52  return shaderSource;
53  }
54 
55  char *shaderSource = new char[len+1];
56 
57  // len isn't always strlen cause some characters are stripped in ascii read...
58  // it is important to 0-terminate the real length later, len is just max possible value...
59  shaderSource[len] = 0;
60  unsigned int i = 0;
61  while (file.good()) {
62  // get character from file.
63  shaderSource[i] = file.get();
64  if (!file.eof()) i++;
65  }
66 
67  shaderSource[i] = 0; // 0 terminate it.
68  file.close();
69  return shaderSource;
70 }
