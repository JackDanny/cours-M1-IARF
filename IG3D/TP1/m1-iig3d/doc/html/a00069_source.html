276 int Renderer::initShader(std::string vsFileName, std::string fsFileName)
277 {
278 
279  // Les opérations devant être faites pour pouvoir utiliser des shaders OpenGL dans notre application sont les suivantes :
280  // 1 - Vertex shader :
281  // 1.1 - Charger le source depuis le fichier "../shaders/vertexdefault.glsl" en utilisant la fonction loaders::text::loadFile() de lecture d'un fichier texte
282  char* vertexShaderSource = loaders::text::loadFile(vsFileName.c_str());
283  // 1.2 - Créer un objet OpenGL VERTEX_SHADER et y associer le source
284  int vertexShaderId;
285  glAssert(vertexShaderId = glCreateShader(GL_VERTEX_SHADER));
286  glAssert(glShaderSource(vertexShaderId, 1, (const GLchar **)&vertexShaderSource, NULL));
287  // 1.3 - Compiler le shader
288  glAssert(glCompileShader(vertexShaderId));
289  // 1.4 - Vérifier les erreurs de compilation
290  GLint compiled;
291  glAssert(glGetShaderiv(vertexShaderId, GL_COMPILE_STATUS, &compiled));
292  if (!compiled) {
293  std::cerr << "[" << vsFileName << "] Vertex shader not compiled : " << std::endl;
294  printShaderInfoLog(vertexShaderId);
295  return 0;
296  }
297 
298  // 2 - Fragment shader :
299  // 2.1 - Charger le source depuis le fichier "../shaders/fragmentdefault.glsl" en utilisant la fonction loaders::text::loadFile() de lecture d'un fichier texte
300  char *fragmentShaderSource;
301  fragmentShaderSource = loaders::text::loadFile(fsFileName.c_str());
302  // 2.2 - Créer un objet OpenGL FRAGMENT_SHADER et y associer le source
303  int fragmentShaderId;
304  glAssert(fragmentShaderId = glCreateShader(GL_FRAGMENT_SHADER));
305  glAssert(glShaderSource(fragmentShaderId, 1, (const GLchar **)&fragmentShaderSource, NULL));
306  // 2.3 - Compiler le shader
307  glAssert(glCompileShader(fragmentShaderId));
308  // 2.4 - Vérifier les erreurs de compilation
309  glAssert(glGetShaderiv(fragmentShaderId, GL_COMPILE_STATUS, &compiled));
310  if (!compiled) {
311  std::cerr << "[" << fsFileName << "] fragment shader not compiled : " << std::endl;
312  printShaderInfoLog(fragmentShaderId);
313  return 0;
314  }
315 
316  // 3 - Programme :
317  // 3.1 - Créer un programme OpenGL, stocker son identifiant dans g_Program et y associer les shaders
318  int program;
319  glAssert(program = glCreateProgram());
320  glAssert(glAttachShader(program, vertexShaderId));
321  glAssert(glAttachShader(program, fragmentShaderId));
322  // 3.2 - Fixer les positions des attributs géométriques en accord avec la classe MyGLMesh et les conventions
323  // d'écriture du source GLSL inPosition --> index 0, inNormal --> index 1 et inTexCoord --> index 2.
324  glAssert(glBindAttribLocation(program, 0, "inPosition"));
325  glAssert(glBindAttribLocation(program, 1, "inNormal"));
326  glAssert(glBindAttribLocation(program, 2, "inTexCoord"));
327  // 3.3 - Lier le programme
328  glAssert(glLinkProgram(program));
329  // 3.4 - Vérifier les erreurs d'édition de lien
330  GLint linked;
331  glAssert(glGetProgramiv(program, GL_LINK_STATUS, &linked));
332  if (!linked) {
333  std::cerr << "Program not linked" << std::endl;
334  printProgramInfoLog(program);
335  }
336  return program;
337 
338 
339 }
